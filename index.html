Simple unit test framework for Arduino.

This is a simple unit test framework for Arduino based on UnitTest++ by Noel Llopis and Charles Nicholson.

Example

An example project for MariaMole is provided here

#include <arduino.h>
#include "Dummy.h"
#include "SerialNotifier.h"

// You must declare this in order to satisfy the grumpy avr-gcc C++ compiler.
extern "C" void   atexit( void ) { } 

SUITE(TestSuite)
{
    TEST(FirstTest)
    {
        int a = 1;
        CHECK(1, a);
        CHECK("bar", "foo");
    }

    TEST(SecondTest)
    {
        CHECK(0, 1);
        CHECK(5, 2);
    }
}

void setup()
{
    Serial.begin(9600);

    Dummy::SerialNotifier notifier;
    Dummy::Result res;

    res = Dummy::Runner::Instance().Run(&notifier); 
}

void loop()
{
}

Once the program is uploaded on the board, you may read the following on your serial terminal. As you may have noticed only 1 of the 4 test is successful.

...o tests\source\SerialTest.cpp:16 TestSuite:FirstTest expected=bar value=foo
...o tests\source\SerialTest.cpp:21 TestSuite:SecondTest expected=0 value=1
...o tests\source\SerialTest.cpp:22 TestSuite:SecondTest expected=5 value=2
Test run: 2, failed: 2, errors: 3

You can also add fixtures by declaring a class with 2 methods Setup and Teardown. These methods will be called respectively before and after the test. Use TEST_FIXTURE to declare a test with a fixture.

class SampleFixture
{
    public:
        void Setup()
        {
            m_device.Open();
            m_device.Setup();
            m_device.WaitUntiReady();
        }
        void TearDown()
        {
            m_device.Close();
        }
    protected:
        ACMEDevice m_device;
}

SUITE(SampleSuite)
{
    TEST_FIXTURE(TaccoTest, SampleFixture)
    {
        uint8_t data;
        uint8_t status;
        status = m_device.GetParameter(ACMEDevice::Tacco, data);
        CHECK(ACMEDevice::CorrectTacco, data);
    }
    TEST_FIXTURE(PizzaTest, SampleFixture)
    {
        char pizzaName[32];
        uint8_t status;
        status = m_device.SetParameter(ACMEDevice::Pizza, "Calzone");
        CHECK(ACMEDevice::OK, status);

        status = m_device.GetParameter(ACMEDevice::Pizza, pizzaName);
        CHECK("Calzone", pizzaName);
    }
}
